    def summary
      if key? 'summary'
        self['summary'].to_s.thru_erb.to_html
      else
        case html.gsub(%r{<(h\d).*?>.*?</\1>}m, '') # omit headings from summary

        # the first HTML block-level element
        when %r{\A\s*<(\w+).*?>.*?</\1>}m
          $&

        # the first paragraph (a run of text delimited by newlines)
        when /\A.*?(?=\r?\n\s*\r?\n)/m
          $&

        end
      end
    end
